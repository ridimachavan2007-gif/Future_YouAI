<script>

let chart;

function analyze() {

  let study = parseInt(document.getElementById("study").value) || 0;

  let screen = parseInt(document.getElementById("screen").value) || 0;

  let sleep = parseInt(document.getElementById("sleep").value) || 0;

  let lang = document.getElementById("lang").value.toLowerCase();

  let hobby = document.getElementById("hobby").value.toLowerCase();

  let domain = document.getElementById("domain").value.toLowerCase();

  let score = study * 12 + sleep * 8 - screen * 10;

  score = Math.max(0, Math.min(100, score));

  let status = "", mindset = "", future = "";

  if (score >= 75) {

    status = "<span class='good'>🌟 Excellent</span>";

    mindset = "You are disciplined. Maintain balance.";

    future = "You are likely to achieve strong career growth.";

  }

  else if (score >= 50) {

    status = "<span class='avg'>⚖️ Average</span>";

    mindset = "Improve consistency for better results.";

    future = "You can reach success with more focus.";

  }

  else {

    status = "<span class='bad'>⚠️ Needs Improvement</span>";

    mindset = "Start small and build habits.";

    future = "Your current habits may slow your growth.";

  }

  let career = "";

  if (domain.includes("ai") || lang.includes("python")) {

    career = "AI Engineer / Data Scientist";

  }

  else if (domain.includes("web") || lang.includes("javascript")) {

    career = "Frontend / Full Stack Developer";

  }

  else if (domain.includes("cyber")) {

    career = "Cybersecurity Analyst";

  }

  else {

    career = "Explore multiple tech domains";

  }

  let hobbyInsight = "";

  if (hobby.includes("gaming")) {

    hobbyInsight = "Gaming improves reflex but control time.";

  }

  else if (hobby.includes("music")) {

    hobbyInsight = "Music boosts creativity and focus.";

  }

  else if (hobby.includes("reading")) {

    hobbyInsight = "Reading improves knowledge and thinking.";

  }

  else {

    hobbyInsight = "Your hobby helps balance your lifestyle.";

  }

  document.getElementById("result").innerHTML = `

    <h3>📊 Analysis</h3>

    <p>Status: ${status}</p>

    <p>Score: ${score}/100</p>

    <div class="bar">

      <div class="fill" id="progress"></div>

    </div>

    <canvas id="myChart"></canvas>

    <h3>🔮 Future</h3>

    <p>${future}</p>

    <h3>🎯 Career</h3>

    <p>${career}</p>

    <h3>🎨 Hobby Insight</h3>

    <p>${hobbyInsight}</p>

    <h3>🧠 Mindset</h3>

    <p>${mindset}</p>

  `;

  setTimeout(() => {

    document.getElementById("progress").style.width = score + "%";

  }, 200);

  if (chart) chart.destroy();

  const ctx = document.getElementById("myChart").getContext("2d");

  chart = new Chart(ctx, {

    type: "bar",

    data: {

      labels: ["Study Hours", "Sleep Hours", "Screen Time"],

      datasets: [{

        label: "Your Habits",

        data: [study, sleep, screen]

      }]

    },

    options: {

      responsive: true

    }

  });

}
